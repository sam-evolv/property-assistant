2. ✅ **Create a new file:** `app/map/page.tsx`
   ```tsx
   'use client';
   import { useEffect, useRef } from 'react';

   export default function MapPage() {
     const mapRef = useRef<HTMLDivElement>(null);

     useEffect(() => {
       const initMap = () => {
         if (!window.google || !mapRef.current) return;
         const map = new window.google.maps.Map(mapRef.current, {
           center: { lat: 51.8979, lng: -8.4711 }, // Cork default
           zoom: 13,
           styles: [
             { elementType: 'geometry', stylers: [{ color: '#0C0D0F' }] },
             { elementType: 'labels.text.fill', stylers: [{ color: '#EDEDED' }] },
           ],
           disableDefaultUI: true,
         });
         new window.google.maps.Marker({
           position: { lat: 51.8979, lng: -8.4711 },
           map,
           title: 'Seaview Development',
         });
       };

       const id = 'google-maps-script';
       if (!document.getElementById(id)) {
         const script = document.createElement('script');
         script.id = id;
         script.src = `https://maps.googleapis.com/maps/api/js?key=${process.env.NEXT_PUBLIC_GOOGLE_MAPS_API_KEY}&libraries=places`;
         script.async = true;
         script.onload = initMap;
         document.body.appendChild(script);
       } else initMap();
     }, []);

     return (
       <div className="h-screen w-full bg-[#0C0D0F]">
         <div ref={mapRef} className="h-full w-full rounded-lg shadow-lg" />
       </div>
     );
   }
