
Kitchen Selections - Complete Feature Build with PC Sums & Analytics Integration
Context
Kitchen selections in Irish property development involve PC (Prime Cost) sums. Purchasers can either:

Take the developer's kitchen/wardrobes - Pay full purchase price
Take their own - PC sum allowance is deducted from purchase price

This directly impacts revenue and needs to be tracked across the platform.

1. Kitchen Selection Options Configuration (Developer Editable)
Each developer should be able to configure their own options in Settings. Store per-tenant:

2. Database Schema
sql-- Kitchen configuration per tenant
CREATE TABLE IF NOT EXISTS kitchen_config (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL REFERENCES tenants(id),
    counter_options JSONB DEFAULT '[]',
    cabinet_options JSONB DEFAULT '[]',
    handle_options JSONB DEFAULT '[]',
    pc_sum_kitchen_4bed DECIMAL(10,2) DEFAULT 7000,
    pc_sum_kitchen_3bed DECIMAL(10,2) DEFAULT 6000,
    pc_sum_kitchen_2bed DECIMAL(10,2) DEFAULT 5000,
    pc_sum_wardrobes DECIMAL(10,2) DEFAULT 1000,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Kitchen selections on units
ALTER TABLE unit_sales_pipeline ADD COLUMN IF NOT EXISTS kitchen_selected BOOLEAN DEFAULT NULL;
ALTER TABLE unit_sales_pipeline ADD COLUMN IF NOT EXISTS kitchen_counter TEXT;
ALTER TABLE unit_sales_pipeline ADD COLUMN IF NOT EXISTS kitchen_cabinet TEXT;
ALTER TABLE unit_sales_pipeline ADD COLUMN IF NOT EXISTS kitchen_handle TEXT;
ALTER TABLE unit_sales_pipeline ADD COLUMN IF NOT EXISTS kitchen_wardrobes BOOLEAN DEFAULT NULL;
ALTER TABLE unit_sales_pipeline ADD COLUMN IF NOT EXISTS kitchen_notes TEXT;
ALTER TABLE unit_sales_pipeline ADD COLUMN IF NOT EXISTS kitchen_date TIMESTAMP;
ALTER TABLE unit_sales_pipeline ADD COLUMN IF NOT EXISTS pc_sum_kitchen DECIMAL(10,2) DEFAULT 0;
ALTER TABLE unit_sales_pipeline ADD COLUMN IF NOT EXISTS pc_sum_wardrobes DECIMAL(10,2) DEFAULT 0;
ALTER TABLE unit_sales_pipeline ADD COLUMN IF NOT EXISTS pc_sum_total DECIMAL(10,2) DEFAULT 0;
